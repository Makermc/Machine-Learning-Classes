最小二乘问题求解

数据来自fred。读取的指标的参数缩写。 运用线性回归预测GDP变化，自变量包括失业率，工人工资和联邦基金率。
# 这是python读取数据并存成csv格式的代码，读取十年间的数据
import pandas_datareader.data as web
from datetime import datetime
start = datetime(2009, 1, 1)
end = datetime(2019, 9, 27)
df = web.DataReader(['A191RL1Q225SBEA', 'UNRATE','PAYEMS','FEDFUNDS'], 'fred', start, end) # 参数看上面的解释
df1 = df[df['A191RL1Q225SBEA'].notna()]
df1.to_csv( 'fred_data.csv',index = None)
freddata = readtable('fred_data.csv');
data = freddata.Variables;
Temp = zscore(data);
Y = Temp(:,1);
X = Temp(:,2:end);
X_h = [X,ones(42,1)];
Beta = (X_h'*X_h)\X_h'*Y;
Pred_Y = X_h*Beta;
RMSE = 1/42*norm(Pred_Y-Y);
RMSE = 0.145 * std(data(:,1)) % 得到zscore前的变化
RMSE = 0.2679
figure(1)
plot(Pred_Y);
hold on
plot(Y)
legend('Predition','Target')
